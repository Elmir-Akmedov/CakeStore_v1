end_day() — add morale system after salary deduction:

python
for w in workers:
    delta = 0
    if net_profit > 0: delta += 2
    if expired_count == 0: delta += 1
    delta -= min(expired_count, 5)
    w.morale = max(0, min(100, w.morale + delta))
    w.save(update_fields=['morale'])
    if w.morale < 20:
        log_event('⚠️', f'{w.name} morale critical!', log_type='warning', is_important=True)
